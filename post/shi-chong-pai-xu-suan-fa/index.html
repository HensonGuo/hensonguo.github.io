八种排序算法

2022-09-08 11 min read

快速排序

快速排序使用分治法（Divide and conquer）策略来把一个串行（list）分为两个子串行（sub-lists）。通过一趟排序将待排记录分割成独立的两部分，则可分别对这两部分记录继续进行排序（递归），以达到整个序列有序。
Ο(n log n)

原理如下：

  1. 选择一个基准元素（pivot），通常选择第一个元素或最后一个元素。
  2. 分区，将待排序序列分成两部分，一部分比基准元素小，另一部分比基准元素大，这个过程称为分区（partition）。
  3. 递归，对两个子序列分别进行快速排序，递归地进行上述过程，直到子序列的长度为1或0，此时子序列已经有序。
  4. 合并，将所有子序列的结果合并起来，得到最终的有序序列。

快速排序的时间复杂度为O(nlogn)，空间复杂度为O(logn)。快速排序是一种原地排序算法，不需要额外的空间来存储中间结果。快速排序是一种高效的排序算法，常用于大数据量的排序场合。

归并排序

归并排序（Merge sort）是建立在归并操作上的一种有效的排序算法。该算法是采用分治法（Divide and Conquer）的一个非常典型的应用。
算法步骤：

  1. 申请空间，使其大小为两个已经排序序列之和，该空间用来存放合并后的序列
  2. 设定两个指针，最初位置分别为两个已经排序序列的起始位置
  3. 比较两个指针所指向的元素，选择相对小的元素放入到合并空间，并移动指针到下一位置
  4. 重复步骤3直到某一指针达到序列尾
  5. 将另一序列剩下的所有元素直接复制到合并序列尾

堆排序

（Heapsort）是指利用堆这种数据结构所设计的一种排序算法。堆积是一个近似完全二叉树的结构，并同时满足堆积的性质：即子结点的键值或索引总是小于（或者大于）它的父节点。
堆排序的平均时间复杂度为Ο(nlogn) 。

//冒泡排序
void bubble_sort(vector<int> &v) {
    int n = v.size();
    bool flag = false;
    for(int i = 0; i < n; ++i) {
        for(int j = 0; j < n - 1 - i; ++j) {
            if(v[j] > v[j + 1]) {
                flag = true;
                swap(v[j], v[j + 1]);
            }
        }
        if(flag == false) {
            break;
        } else {
            flag = true;
        }
    }
}

//选择排序
void select_sort(vector<int> &v) {
    int n = v.size();
    for(int i = 0; i < n - 1; ++i) {
        int min_index = i;
        for(int j = i + 1; j < n; ++j) {
            if(v[min_index] > v[j]) {
                min_index = j;
            }
        }
        if(min_index != i) {
            swap(v[min_index], v[i]);
        }
    }
}

//插入排序
void inser_sort(vector<int> &v) {
    int n = v.size();
    for(int i = 1; i < n; ++i) {
        int insert_val = v[i];
        int insert_index = i - 1;
        while(insert_index >= 0 && insert_val < v[insert_index]) {
            v[insert_index + 1] = v[insert_index];
            --insert_index;
        }
        if(insert_index + 1 != i) {
            v[insert_index + 1] = insert_val;
        }
    }
}

//希尔排序--交换法
void shell_sort1(vector<int> &v) {
    int n = v.size();
    for(int gap = n / 2; gap > 0; gap /= 2) {
        for(int i = gap; i < n; ++i) {
            for(int j = i - gap; j >= 0; j -= gap) {
                if(v[j] > v[j + gap]) {
                    swap(v[j], v[j + gap]);
                }
            }
        }
    }
}

//希尔排序--移步法
void shell_sort2(vector<int> &v) {
    int n = v.size();
    for(int gap = n / 2; gap > 0; gap /= 2) {
        for(int i = gap; i < n; ++i) {
            int j = i;
            int tmp = v[j];
            if(v[j] < v[j - gap]) {
                while(j - gap >= 0 && tmp < v[j - gap]) {
                    v[j] = v[j - gap];
                    j -= gap;
                }
                v[j] = tmp;
            }
        }
    }
}

//快速排序
void quick_sort(vector<int> &v, int left, int right) {
    int l = left;
    int r = right;
    int mid = left + (right - left) / 2;
    while(l < r) {
        while(v[l] < v[mid]) {
            ++l;
        }
        while(v[r] > v[mid]) {
            --r;
        }
        if(l >= r) {
            break;
        }
        swap(v[l], v[r]);
        if(v[l] == v[mid]) {
            --r;
        }
        if(v[r] == v[mid]) {
            ++l;
        }
    }
    if(l == r) {
        ++l;
        --r;
    }
    if(left < r) {
        quick_sort(v, left, r);
    }
    if(right > l) {
        quick_sort(v, l, right);
    }
}

//归并排序
void merge(vector<int> &v, int left, int right, int mid, vector<int> &tmp) {
    int i = left;
    int j = mid + 1;
    int t = 0;
    while(i <= mid && j <= right) {
        if(v[i] <= v[j]) {
            tmp[t] = v[i];
            ++t;
            ++i;
        } else {
            tmp[t] = v[j];
            ++t;
            ++j;
        }
    }
    while(i <= mid) {
        tmp[t] = v[i];
        ++t;
        ++i;
    }
    while(j <= right) {
        tmp[t] = v[j];
        ++t;
        ++i;
    }
    t = 0;
    int tmpleft = left;
    while(tmpleft <= right) {
        v[tmpleft] = tmp[t];
        ++t;
        ++tmpleft;
    }
}

void merge_sort(vector<int> &v, int left, int right, vector<int> &tmp) {
    if(left < right) {
        int mid = left + (right - left) / 2;
        merge_sort(v, left, mid, tmp);
        merge_sort(v, mid + 1, right, tmp);
        merge(v, left, right, mid, tmp);
    }
}

//基数排序
int maxbit(vector<int> &v) {
    int d = 1;
    int p = 10;
    int n = v.size();
    for(int i = 0; i < n; ++i) {
        while(v[i] >= p) {
            p *= 10;
            ++d;
        }
    }
    return d;
}

void radix_sort(vector<int> &v) {
    int d = maxbit(v);
    int n = v.size();
    int i = 0, j = 0, k = 0, radix = 1;
    vector<int> tmp(n, 0), count(10, 0);
    for(i = 1; i <= d; ++i) {
        for(j = 0; j < 10; ++j) {
            count[j] = 0;
        }
        for(j = 0; j < n; ++j) {
            k = (v[j] / radix) % 10;
            ++count[k];
        }
        for(j = 1; j < 10; ++j) {
            count[j] = count[j - 1] + count[j];
        }
        for(j = n - 1; j >= 0; --j) {
            k = (v[j] / radix) % 10;
            tmp[count[k] - 1] = v[j];
            --count[k];
        }
        for(j = 0; j < n; ++j) {
            v[j] = tmp[j];
        }
        radix *= 10;
    }
}

//桶排序
void bucket_sort(vector<int> &v) {
    int min_val = 0;
    int max_val = 0;
    int n = v.size();
    for(int i = 0; i < n; ++i) {
        if(v[i] > max_val) {
            max_val = v[i];
        }
        if(v[i] < min_val) {
            min_val = v[i];
        }
    }
    vector<int> tmp(n, 0), count(max_val - min_val + 1, 0);
    for(int i = 0; i < n; ++i) {
        int pos = v[i] - min_val;
        ++count[pos];
    }
    n = max_val - min_val + 1;
    for(int i = 1; i < n; ++i) {
        count[i] = count[i - 1] + count[i];
    }
    n = v.size();
    for(int i = n - 1; i >= 0; --i) {
        int pos = v[i] - min_val;
        tmp[count[pos] - 1] = v[i];
        --count[pos];
    }
    for(int i = 0; i < n; ++i) {
        v[i] = tmp[i];
    }
}

//堆排序
void adjust_heap(vector<int> &v, int i, int n) {
    for(int k = i * 2 + 1; i < n; k = k * 2 + 1) {
        if(k + 1 < n && v[k] < v[k + 1]) {
            ++k;
        }
        if(k < n && v[k] > v[i]) {
            swap(v[i], v[k]);
            i = k;
        } else {
            break;
        }
    }
}

void heap_sort(vector<int> &v) {
    int n = v.size();
    for(int i = n / 2 - 1; i >= 0; --i) {
        adjust_heap(v, i, n);
    }
    for(int j = n - 1; j > 0; --j) {
        swap(v[j], v[0]);
        adjust_heap(v, 0, j);
    }
}

8大排序算法排名

排序算法的效率和适用场景有很多因素影响，因此没有一个固定的排名。以下是8大排序算法的简介和特点：

  1. 冒泡排序（Bubble Sort）：简单但效率较低的排序算法，时间复杂度为O(n^2)，适用于小规模数据的排序。
  2. 选择排序（Selection Sort）：简单但效率较低的排序算法，时间复杂度为O(n^2)，适用于小规模数据的排序。
  3. 插入排序（Insertion Sort）：简单但效率较高的排序算法，时间复杂度为O(n^2)，适用于小规模数据的排序。
  4. 希尔排序（Shell Sort）：基于插入排序的一种排序算法，时间复杂度为O(nlogn)~O(n^2)，适用于中等规模数据的排序。
  5. 归并排序（Merge Sort）：基于分治思想的一种排序算法，时间复杂度为O(nlogn)，适用于大规模数据的排序，但需要额外的空间来存储中间结果。
  6. 快速排序（Quick Sort）：基于分治思想的一种排序算法，时间复杂度为O(nlogn)，适用于大规模数据的排序，但在最坏情况下时间复杂度为O(n^2)。
  7. 堆排序（Heap Sort）：基于堆的一种排序算法，时间复杂度为O(nlogn)，适用于大规模数据的排序，但需要额外的空间来存储堆。
  8. 基数排序（Radix Sort）：基于数字位的一种排序算法，时间复杂度为O(d(n+r))，适用于大规模数据的排序，但需要额外的空间来存储桶和中间结果，只适用于整数排序。

综合来看，归并排序和快速排序在大规模数据的排序中表现较好，而插入排序和希尔排序在小规模数据的排序中表现较好。但在实际应用中，需要根据具体情况选择合适的排序算法。
